## Setup

```typescript
const BASE_URL = process.env.FELLOW_API_URL ?? 'https://api.fellowpay.com.br';
const API_KEY = process.env.FELLOW_API_KEY!;

async function fellowCore(path: string, options: RequestInit = {}) {
  const res = await fetch(`${BASE_URL}${path}`, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      'X-Api-Key': API_KEY,
      ...options.headers,
    },
  });

  if (!res.ok) {
    const error = await res.json();
    throw new Error(`[${error.status}] ${error.title}: ${error.detail}`);
  }

  return res.json();
}
```

## Receber webhook (Express)

```typescript
import express from 'express';
import crypto from 'node:crypto';

const app = express();

// IMPORTANTE: usar raw body para validar HMAC
app.use('/webhooks', express.raw({ type: 'application/json' }));

app.post('/webhooks/fellow', (req, res) => {
  const signature = req.headers['x-fellow-signature'] as string;
  const secret = process.env.WEBHOOK_SECRET!;

  // Validar assinatura
  const expected = crypto
    .createHmac('sha256', secret)
    .update(req.body)
    .digest('hex');

  const received = signature?.replace('sha256=', '') ?? '';

  if (!crypto.timingSafeEqual(
    Buffer.from(expected, 'hex'),
    Buffer.from(received, 'hex')
  )) {
    return res.status(401).send('Invalid signature');
  }

  const event = JSON.parse(req.body.toString());

  switch (event.event) {
    case 'transaction.captured':
      console.log('Pagamento confirmado:', event.data.transactionId);
      // Atualizar pedido no banco...
      break;

    case 'transaction.refunded':
      console.log('Reembolso:', event.data.transactionId);
      break;

    case 'seller.approved':
      console.log('Seller aprovado:', event.data.sellerId);
      break;
  }

  res.status(200).send('OK');
});

app.listen(3000);
```
